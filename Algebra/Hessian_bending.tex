\LARGE\textbf{Bending Energy Gradient and Hessian  \\}
\vspace{3 mm}
\end{center}


The bending energy can be written as 
\begin{equation}
    E_b = \kappa \displaystyle\int_{\mathcal{M}} (H-H_0)^2dA
\end{equation}

In out discrete setting we write it as  

\begin{equation}
    E_b = \kappa \sum_{v_i} \left( \cfrac{H_i}{A_i} - H_0 \right)^2 A_i   
\end{equation}


Using this convention we can write the gradient of the total bending energy as 

\begin{align}
    \cfrac{\nabla E_b}{\kappa} &=  \displaystyle\sum_{v_i} \nabla  \left[\left( \cfrac{H_i}{A_i} - H_0 \right)^2 A_i \right] \\
    &=  \sum_{v_i} \left( \cfrac{H_i}{A_i}-H_0 \right)^2\nabla A_i + 2A_i  \left(\cfrac{H_i}{A_i}-H_0 \right)\left[ \cfrac{\nabla H_i}{A_i}-\cfrac{H_i}{A_i^2} \nabla A_i \right]\\
    &= \sum_{v_i}\left[ \left(\cfrac{H_i}{A_i}-H_0 \right)\left(  \cfrac{H_i}{A_i}-H_0 - 2\cfrac{H_i}{A_i}  \right) \right] \nabla A_i + 2\left( \cfrac{H_i}{A}-H_0\right)\nabla H_i\\
    &=\sum_{v_i} -\left[\cfrac{H_i^2}{A_i^2}-H_0^2  \right]\nabla A_i + 2 \left(\cfrac{H_i}{A_i}-H_0  \right)\nabla H_i
\end{align}

Now we just need to find the definitions for the gradients of the dual area and the gradient of the mean curvature and we win (:

For the dual area depends on your convention i use the barycentric dual area, if we call the area for triangle $j$ as $T_j$
\begin{align}
    A_i &= \sum_{f_j \ni v_i } \cfrac{T_j}{3}\\
    \implies \grad A_i &= \cfrac{1}{3} \sum_{f_j \ni v_i} \nabla T_j
\end{align}

And for the gradient of the triangle $j$. If we define $u= p_2-p_1$, $v = p_3-p_1$. using the comma notation for derivatives $u_{,p_1} = -id$, $u_{,p_2}= id$, $v_{,p_1} = -id$, $v_{,p_3} = id$.
Then we can define $z$ as  $z= u\times v$. such that  $z_{,u} = [u]_\times$ and $z_u = - [v]_\times$ .

Then 
\begin{align}
    A &= \norm||z\norm||\\
    A_{,z} &= \cfrac{z^T}{\norm|| z\norm||}\\
    \implies A_{,u} &= A_{,z}z_{,u} = -\cfrac{1}{\norm|| z\norm||} z^T[v]_\times = -\,\cfrac{1}{\norm||z\norm||} \left( [v]_\times^T z \right)^T\\
    &=\cfrac{1}{\norm||z\norm||} (v\times z)^T \\
    A_{,v} &= A_{,z}z_{,v} = \cfrac{1}{\norm ||z\norm||} z^T [u]_\times = \cfrac{1}{\norm||z\norm||}([u]_{\times}^Tz)^T \\
    &=- \cfrac{1}{\norm|| z \norm||}(u\times z)^T
\end{align}

Using this calculation we can write the derivative of the area w.r.t the 3 points.

\begin{align}
    A_{,p1} &= A_{,u}u_{,p_1}+ A_{,v}v_{,p_1} = -A_{,u}-A_{,v}\\
    A_{,p2} &= A_{,u}u_{,p_2} = A_{,u}\\
    A_{,p3} &= A_{,v}v_{,p_3} = A_{,v}
\end{align}

Now we 'just' need to do the gradient of the $H_i$

For this it will depend on the discretization of the mean curvature you use. THe one i have implemented can be written as 

\begin{equation}
    H_i = \sum_{e_{ij}\ni v_i} \cfrac{\ell_{ij}\varphi_{ij}}{4}
\end{equation}

You can writhe the gradient of this term as 
\begin{align}
    \nabla H_i = \sum_{e_{ij}\ni v_i }\cfrac{1}{4}\left( \varphi_{ij}\nabla\ell_{ij} + \ell_{ij}\nabla\varphi_{ij}   \right)
\end{align}
Now the interesting thing here is that $\ell_{ij}$ depends on two points so the gradient is a $6\times1$ vector and the dihedral depends on 4 vertices so the gradient is a $12\times 1$ vector.


The gradient of $\ell_{ij}$ is simple, for an edge $u = p1-p2$ 
\begin{align}
    \nabla_{p_1} ||u|| = \hat{u}\\
    \nabla_{p_2} ||u|| = - \hat{u}  
\end{align}

And this is where the fun ends. 

Now we need to calculate the gradient of the dihedral angle $\varphi_{ij}$

We start considering a triangle flap, that is, two triangles connected by an edge. We name the point $p_1$ as the lower one belonging to the edge. $p_2$ as the other point that lies on the edge connecting the flap. $p_3$ to the right and $p_4$ to the left.

We can define the dihedral angle as 

\begin{align}
    \varphi &= \mathrm{atan2}\left( \left\Vert \cfrac{x\times z}{\Vert x\times z\Vert} \times \cfrac{y\times x}{\Verty\times x\Vert}  \right\Vert , \cfrac{x\times z}{\Vert x\times z\Vert }\cdot \cfrac{y\times x}{\Vert y\times x\Vert}     \right)\\
            &= \mathrm{atan2}\left( \Vert (x\times z)\times (x\times y) \Vert,(x\times z)\cdot (y\times x)  \right)\\
            &= \mathrm{atan2}\left(\Vert \det(x,y,z) \cdot x\Vert, (x\times z)\cdot (y\times x)  \right)\\
            &= \mathrm{atan2}\left( \det(x,y,z)\Vert x\Vert,(x\times z)\cdot (y\times x)  \right)
\end{align}

That is out silly little expression for the dihedral. We will use $u,v,w$ instead of $x,y,z$ but thats how it was on the image :p. 

Defining $u = p_2-p_1$, $v= p_3-p_1$ and $w = p_4-p_1$.

And the functions  $g :=\det(u,v,w)\Vert u \Vert$, and $h = (v\times w)\cdot (v\times u)$ and $r = g^2+h^2$. We get.

\begin{equation}
    f = \mathrm{atan2}(g,h)
\end{equation}
We can rewrite $h$ as 
\begin{align}
    h &= u^T[w]_\times [v]_{\times} u \\
    &= u^T[v]_\times [w]_\times u\\
    &= w^T [u]_\times^2 v \\
    &= v^T [u]_\times^2 w
\end{align}

Now we need to do the derivatives. Because i am lazy i will not write the coma and the subindices here means derivative w.r.t that variable.

\begin{align}
    f_u = f_h h_u + f_g g_u &= -\cfrac{g}{r} u^T([w]_\times [v]_\times +[v]_\times[w]_\times )+ \cfrac{h}{r}\left( (v\times w)^T\Vert u\Vert + \det(u,v,w)\cfrac{u^T}{\Vert u \Vert }  \right)\\
    &=\cfrac{1}{r}\left(-g u^T ([w]_\times [v]_\times +[v]_\times [w]_\times)  + h (\Vert u\Vert (u\times w)^T + \det(u,v,w)\cfrac{u^T}{\Vert u\Vert })  \right)
\end{align}

And the other two 

\begin{align}
    f_v = f_h h_v + f_gg_v &= \cfrac{1}{r}(-gh_v + hg_v)\\
    &=\cfrac{1}{r}\left( -g w^T [u]_\times^2 +h\Vert u \Vert (w\times u)^T\right)
\end{align}

\begin{align}
    f_w = f_h h_w f_gg_w &= \cfrac{1}{r}\left( -gh_w +hg_w \right)\\
    &=\cfrac{1}{r}\left(-gv^T [u]_\times^2 + h\Vert u \Vert u (u\times v)^T \right)
\end{align}

Then we have 
\begin{equation}
    f_{p_1} = -f_u -f_v -f_w \quad f_{p_2} = f_u \quad f_{p_3} = f_v \quad f_{p_4} = f_w
\end{equation}
